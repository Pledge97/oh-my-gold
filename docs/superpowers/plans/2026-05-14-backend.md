# backend/core/event_bus.py
from collections import defaultdict
from typing import Callable, Any


class EventBus:
    def __init__(self):
        self._handlers: dict[str, list[Callable]] = defaultdict(list)

    def subscribe(self, event: str, handler: Callable) -> None:
        self._handlers[event].append(handler)

    def unsubscribe(self, event: str, handler: Callable) -> None:
        self._handlers[event] = [h for h in self._handlers[event] if h != handler]

    def publish(self, event: str, data: Any) -> None:
        for handler in list(self._handlers[event]):
            handler(data)


bus = EventBus()  # 全局单例
```

# backend/data/kline.py
import pandas as pd

_COLS = ["ts", "open", "high", "low", "close", "volume"]


def build_kline(ticks: list[dict], period_sec: int) -> pd.DataFrame:
    if not ticks:
        return pd.DataFrame(columns=_COLS)
    df = pd.DataFrame(ticks)
    df["dt"] = pd.to_datetime(df["ts"], unit="ms", utc=True)
    df = df.set_index("dt").sort_index()
    rule = f"{period_sec}s"
    ohlc = df["price"].resample(rule).ohlc().dropna()
    ohlc["volume"] = df["price"].resample(rule).count()
    ohlc = ohlc.reset_index()
    ohlc["ts"] = (ohlc["dt"].astype("int64") // 10**6).astype(int)
    return ohlc[_COLS].reset_index(drop=True)
```

import pandas as pd

@dataclass

    def update(self, snapshot: IndicatorSnapshot) -> None:

    def get(self) -> IndicatorSnapshot:

# backend/data/fetcher_tick.py
import time
import akshare as ak
from backend.db.database import get_conn
from backend import config


def fetch_akshare_price() -> float:
    df = ak.spot_quotation_sge()
    row = df[df["品种"] == config.AKSHARE_SYMBOL].iloc[0]
    return float(row["最新价"])


def store_tick(ts: int, price: float) -> None:
    with get_conn() as conn:
        conn.execute("INSERT INTO prices (ts, price) VALUES (?, ?)", (ts, price))


def run_once() -> float | None:
    try:
        price = fetch_akshare_price()
        ts = int(time.time() * 1000)
        store_tick(ts, price)
        return price
    except Exception as e:
        print(f"[fetcher_tick] 采集失败: {e}")
        return None
```

import pandas as pd
from backend.core.enums import MarketState
from backend import config


def detect_regime(adx_result: dict) -> MarketState:
    adx = adx_result["adx"]
    plus_di = adx_result["plus_di"]
    minus_di = adx_result["minus_di"]
    adx_series: pd.Series = adx_result["adx_series"]

    if adx < config.ADX_TREND_THRESHOLD:
        return MarketState.OSCILLATION

    lookback = config.ADX_LOOKBACK
    adx_prev = float(adx_series.iloc[-(lookback + 1)]) if len(adx_series) > lookback else adx
    if adx <= adx_prev:
        return MarketState.TREND_DECAY

    return MarketState.TREND_UP if plus_di > minus_di else MarketState.TREND_DOWN

# backend/signals/exit_signal.py
from backend.core.enums import MarketState, SignalType, TradingMode
from backend import config


def check_exit(open_price: float, current_price: float, peak_price: float,
               atr: float, bb_upper: float, ema20_5min: float,
               mode: TradingMode, market_state: MarketState) -> SignalType:
    profit_rate = (current_price - open_price) / open_price
    stop_price = open_price - config.OSC_STOP_LOSS_ATR_MULT * atr
    fixed_stop = open_price * (1 - config.TREND_STOP_LOSS_FIXED)

    if current_price <= max(stop_price, fixed_stop):
        return SignalType.STOP_LOSS

    if mode == TradingMode.OSCILLATION:
        if profit_rate >= config.OSC_TAKE_PROFIT_RATE:
            return SignalType.TAKE_PROFIT
        if current_price >= bb_upper and profit_rate >= config.SELL_FEE_RATE:
            return SignalType.TAKE_PROFIT

    if mode == TradingMode.TREND:
        if market_state == MarketState.TREND_DECAY:
            trail = peak_price * (1 - config.TREND_DECAY_TRAIL)
            if current_price <= trail and profit_rate > 0:
                return SignalType.TAKE_PROFIT
        if current_price < ema20_5min and profit_rate >= config.TREND_MIN_PROFIT:
            return SignalType.TAKE_PROFIT

    return SignalType.NONE

# backend/risk/position.py
from backend.core.enums import TradingMode, CloseType
from backend import config


class PositionManager:
    def __init__(self, t_position_g: float = config.T_POSITION_G):
        self._t_position_g = t_position_g
        self._next_id = 1
        self._positions: dict[int, dict] = {}

    def total_open_g(self) -> float:
        return sum(p["amount_g"] for p in self._positions.values() if p["status"] == "OPEN")

    def open(self, open_price: float, amount_g: float,
             ts: int, mode: TradingMode) -> int | None:
        if self.total_open_g() + amount_g > self._t_position_g:
            return None
        pid = self._next_id
        self._next_id += 1
        self._positions[pid] = {
            "id": pid, "open_ts": ts, "open_price": open_price,
            "amount_g": amount_g, "add_count": 0, "status": "OPEN",
            "mode": mode.value, "peak_price": open_price,
        }
        return pid

    def add(self, pid: int, amount_g: float) -> None:
        pos = self._positions[pid]
        pos["amount_g"] += amount_g
        pos["add_count"] += 1

    def update_peak(self, pid: int, price: float) -> None:
        pos = self._positions[pid]
        if price > pos["peak_price"]:
            pos["peak_price"] = price

    def close(self, pid: int, close_price: float,
              ts: int, close_type: CloseType) -> dict:
        pos = self._positions[pid]
        fee = close_price * pos["amount_g"] * config.SELL_FEE_RATE
        pnl_yuan = (close_price - pos["open_price"]) * pos["amount_g"] - fee
        pnl_g = pnl_yuan / close_price
        pos.update({"status": "CLOSED", "close_ts": ts, "close_price": close_price,
                    "close_type": close_type.value, "pnl_yuan": pnl_yuan, "pnl_g": pnl_g})
        return {"pnl_yuan": pnl_yuan, "pnl_g": pnl_g}

    def get(self, pid: int) -> dict:
        return self._positions[pid]

    def open_positions(self) -> list[dict]:
        return [p for p in self._positions.values() if p["status"] == "OPEN"]
```

# backend/risk/circuit_breaker.py
from backend import config


class CircuitBreaker:
    def __init__(self):
        self.level: int | None = None
        self._resume_ts: int | None = None
        self.daily_stop_count: int = 0

    def is_active(self) -> bool:
        return self.level is not None

    def _activate(self, level: int, pause_ms: int | None, ts: int) -> None:
        if self.level is None or level > self.level:
            self.level = level
            self._resume_ts = (ts + pause_ms) if pause_ms else None

    def check_tick(self, prev_price: float, current_price: float, ts: int) -> None:
        if prev_price == 0:
            return
        pct = abs(current_price - prev_price) / prev_price
        if pct >= config.CB1_TICK_PCT:
            self._activate(1, config.CB1_TICK_PAUSE_MIN * 60 * 1000, ts)

    def check_5min_range(self, high: float, low: float, ts: int) -> None:
        if low == 0:
            return
        pct = (high - low) / low
        if pct >= config.CB1_5MIN_PCT:
            self._activate(1, config.CB1_5MIN_PAUSE_MIN * 60 * 1000, ts)

    def check_atr(self, current_atr: float, historical_mean: float, ts: int) -> None:
        if historical_mean == 0:
            return
        if current_atr / historical_mean >= config.CB2_ATR_MULT:
            self._activate(2, None, ts)

    def record_stop_loss(self, ts: int) -> None:
        self.daily_stop_count += 1
        if self.daily_stop_count >= config.CB3_DAILY_STOP_COUNT:
            self._activate(3, None, ts)

    def check_resume(self, current_ts: int) -> None:
        if self.level == 1 and self._resume_ts and current_ts >= self._resume_ts:
            self.level = None
            self._resume_ts = None

    def manual_resume(self) -> None:
        if self.level == 3:
            self.level = None

    def reset_daily(self) -> None:
        self.daily_stop_count = 0
        if self.level in (1, 2):
            self.level = None
```

# backend/risk/risk_manager.py
from backend import config


class RiskManager:
    def __init__(self, t_position_g: float = config.T_POSITION_G,
                 gold_price: float = 1000.0):
        self._t_g = t_position_g
        self._gold_price = gold_price
        self._daily_pnl = 0.0
        self._halted = False
        self._loss_days = 0
        self._profit_days = 0

    def _limit(self) -> float:
        return self._t_g * self._gold_price * config.DAILY_LOSS_LIMIT_RATE

    def record_pnl(self, pnl_yuan: float) -> None:
        self._daily_pnl += pnl_yuan
        if self._daily_pnl <= -self._limit():
            self._halted = True

    def can_trade(self) -> bool:
        return not self._halted

    def reset_daily(self, gold_price: float) -> None:
        self._gold_price = gold_price
        if self._daily_pnl < 0:
            self._loss_days += 1
            self._profit_days = 0
        else:
            self._profit_days += 1
            if self._profit_days >= config.CONSECUTIVE_LOSS_DAYS:
                self._loss_days = 0
        self._daily_pnl = 0.0
        self._halted = False

    def record_daily_loss(self) -> None:
        self._loss_days += 1
        self._profit_days = 0

    def record_daily_profit(self) -> None:
        self._profit_days += 1
        if self._profit_days >= config.CONSECUTIVE_LOSS_DAYS:
            self._loss_days = 0

    def unit_buy_g(self) -> float:
        return config.REDUCED_UNIT_BUY_G if self._loss_days >= config.CONSECUTIVE_LOSS_DAYS \
               else config.UNIT_BUY_G

# backend/strategy/engine.py
import time
import pandas as pd
from backend.core.enums import MarketState, SignalType, TradingMode, CloseType
from backend.core.context import context, IndicatorSnapshot
from backend.core.event_bus import bus
from backend.risk.circuit_breaker import CircuitBreaker

from backend.signals.buy_signal import check_buy
from backend.signals.exit_signal import check_exit
from backend.db.database import get_conn
from backend import config


class StrategyEngine:
    def __init__(self):
        self.positions = PositionManager()
        self.risk = RiskManager()
        self.cb = CircuitBreaker()
        self._prev_price: float | None = None

    def on_tick(self, data: dict) -> dict:
        price: float = data["price"]
        ts: int = data["ts"]
        snap: IndicatorSnapshot = context.get()

        if self._prev_price is not None:
            self.cb.check_tick(self._prev_price, price, ts)
        self.cb.check_resume(ts)

        self._check_exits(price, ts, snap)

        result = {
            "ts": ts, "price": price,
            "market_state": snap.market_state.value,
            "indicators": {
                "adx": snap.adx, "plus_di": snap.plus_di, "minus_di": snap.minus_di,
                "bb_upper": snap.bb_upper, "bb_mid": snap.bb_mid, "bb_lower": snap.bb_lower,
                "rsi": snap.rsi, "atr": snap.atr_5min,
            },
            "signal": None,
            "circuit_breaker": {"active": self.cb.is_active(), "level": self.cb.level},
            "positions": self.positions.open_positions(),
        }

        if not self.cb.is_active() and self.risk.can_trade() and snap.ready:
            sig = check_buy(snap)
            if sig == SignalType.BUY:
                unit_g = self.risk.unit_buy_g()
                mode = (TradingMode.TREND if snap.market_state == MarketState.TREND_UP
                        else TradingMode.OSCILLATION)
                pid = self.positions.open(price, unit_g, ts, mode)
                if pid:
                    reason = (f"RSI回调至{snap.rsi:.1f}，趋势上涨确认"
                              if mode == TradingMode.TREND
                              else f"价格触及布林下轨，RSI={snap.rsi:.1f}")
                    self._save_signal(ts, "BUY", mode.value, price, unit_g, reason)
                    result["signal"] = {"type": "BUY", "amount_g": unit_g, "reason": reason}

        self._prev_price = price
        return result

    def _check_exits(self, price: float, ts: int, snap: IndicatorSnapshot) -> None:
        for pos in self.positions.open_positions():
            self.positions.update_peak(pos["id"], price)
            mode = TradingMode(pos["mode"])
            sig = check_exit(
                open_price=pos["open_price"], current_price=price,
                peak_price=pos["peak_price"], atr=snap.atr_5min,
                bb_upper=snap.bb_upper, ema20_5min=snap.ema20_5min,
                mode=mode, market_state=snap.market_state,
            )
            if sig in (SignalType.TAKE_PROFIT, SignalType.STOP_LOSS):
                close_type = CloseType.TAKE_PROFIT if sig == SignalType.TAKE_PROFIT \
                             else CloseType.STOP_LOSS
                pnl = self.positions.close(pos["id"], price, ts, close_type)
                self.risk.record_pnl(pnl["pnl_yuan"])
                if close_type == CloseType.STOP_LOSS:
                    self.cb.record_stop_loss(ts)
                self._save_signal(ts, close_type.value, pos["mode"], price,
                                  pos["amount_g"], f"信号:{sig.value}")

    def _save_signal(self, ts, sig_type, mode, price, amount_g, reason):
        with get_conn() as conn:
            conn.execute(
                "INSERT INTO signals (ts, type, mode, price, amount_g, reason) "
                "VALUES (?, ?, ?, ?, ?, ?)",
                (ts, sig_type, mode, price, amount_g, reason),
            )

            "SELECT ts, price FROM prices ORDER BY ts DESC LIMIT ?", (limit,)
        ).fetchall()
    return [{"ts": r["ts"], "price": r["price"]} for r in reversed(rows)]


def _load_daily_df() -> pd.DataFrame:
    with get_conn() as conn:
        rows = conn.execute(
            "SELECT open, high, low, close FROM daily_prices ORDER BY date ASC"
        ).fetchall()
    return pd.DataFrame([dict(r) for r in rows])


async def tick_job(engine, broadcast_fn):
    price = fetch_tick()
    if price is None:
        return

    # 静默采集jdjygold
    try:
        async with httpx.AsyncClient(timeout=4) as client:
            resp = await client.get(config.JDJYGOLD_URL,
                                    params={"productSku": config.JDJYGOLD_SKU})
            d = resp.json()
            if d.get("success"):
                store_tick(int(d["resultData"]["datas"]["time"]),
                           float(d["resultData"]["datas"]["price"]))
    except Exception:
        pass

    # 更新context
    ticks_5min = _load_ticks(500 * 60)
    ticks_4h   = _load_ticks(100 * 2880)
    kline_5min = build_kline(ticks_5min, period_sec=300)
    kline_4h   = build_kline(ticks_4h,   period_sec=14400)
    daily_df   = _load_daily_df()

    snap = IndicatorSnapshot(price=price, ts=int(time.time() * 1000))
    min_daily = config.ADX_PERIOD + config.ADX_LOOKBACK
    min_5min  = max(config.BB_PERIOD, config.ATR_PERIOD, config.RSI_PERIOD)

    if len(daily_df) >= min_daily and not kline_5min.empty and len(kline_5min) >= min_5min:
        adx_res = calc_adx(daily_df, config.ADX_PERIOD)
        snap.adx       = adx_res["adx"]
        snap.plus_di   = adx_res["plus_di"]
        snap.minus_di  = adx_res["minus_di"]
        snap.adx_series = adx_res["adx_series"]
        snap.market_state = detect_regime(adx_res)

        bb = calc_bollinger(kline_5min, config.BB_PERIOD, config.BB_STD)
        snap.bb_upper  = bb["upper"]
        snap.bb_mid    = bb["mid"]
        snap.bb_lower  = bb["lower"]
        snap.rsi       = calc_rsi(kline_5min, config.RSI_PERIOD)
        snap.atr_5min  = calc_atr(kline_5min, config.ATR_PERIOD)
        snap.ema20_5min = float(calc_ema(kline_5min, config.EMA_SHORT).iloc[-1])

        if not kline_4h.empty and len(kline_4h) >= config.EMA_LONG:
            snap.ema20_4h = float(calc_ema(kline_4h, config.EMA_SHORT).iloc[-1])
            snap.ema60_4h = float(calc_ema(kline_4h, config.EMA_LONG).iloc[-1])

        snap.ready = True

    context.update(snap)
    bus.publish("tick", {"price": price, "ts": snap.ts})

    result = engine.on_tick({"price": price, "ts": snap.ts})
    await broadcast_fn(result)

# backend/api/websocket.py
from fastapi import WebSocket, WebSocketDisconnect
from typing import Any

_clients: list[WebSocket] = []


async def ws_endpoint(websocket: WebSocket):
    await websocket.accept()
    _clients.append(websocket)
    try:
        while True:
            await websocket.receive_text()
    except WebSocketDisconnect:
        if websocket in _clients:
            _clients.remove(websocket)


async def broadcast(data: Any):
    dead = []
    for client in _clients:
        try:
            await client.send_json(data)
        except Exception:
            dead.append(client)
    for c in dead:
        _clients.remove(c)
```

from contextlib import asynccontextmanager
from fastapi import FastAPI, WebSocket
from apscheduler.schedulers.asyncio import AsyncIOScheduler
from backend.db.database import init_db
from backend.data.fetch_history import fetch_and_store
from backend.strategy.engine import StrategyEngine

from backend.api.websocket import ws_endpoint, broadcast
from backend.core.scheduler import tick_job
from backend import config

engine = StrategyEngine()
scheduler = AsyncIOScheduler()


@asynccontextmanager
async def lifespan(app: FastAPI):
    init_db()
    fetch_and_store()
    scheduler.add_job(
        tick_job, "interval", seconds=config.TICK_INTERVAL_SEC,
        kwargs={"engine": engine, "broadcast_fn": broadcast}
    )
    scheduler.start()
    yield
    scheduler.shutdown()


app = FastAPI(lifespan=lifespan)
app.include_router(router)


@app.websocket("/ws")
async def websocket_route(websocket: WebSocket):
    await ws_endpoint(websocket)
