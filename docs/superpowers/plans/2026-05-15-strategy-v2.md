# 策略 V2.0 — 组合仓位管理实现计划

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** 将后端策略从 v1 单笔独立仓位模式，升级为 v2 组合仓位管理：分批建仓（50g/30g/20g）、组合止盈（60%/20%/清仓）、组合止损（-1.5%/-2.5%/-3.5%），同时更新前端持仓表展示组合维度数据。

**Architecture:** 重点改造三层：`risk/position.py`（组合持仓核心）、`signals/`（buy/sell/exit 三个信号模块用组合盈亏率替换单笔逻辑）、`strategy/engine.py`（信号流程新增T仓整体盈亏计算步骤）。数据库新增 `position_lots` 表记录每批次买入明细，`positions` 表升级为记录整轮（从初始建仓到清仓）汇总数据。前端 `PositionTable.tsx` 展示组合持仓视图。

**Tech Stack:** Python 3.11, FastAPI, SQLite, pandas, pytest; React 18 + TypeScript + Ant Design

```python
# backend/risk/position.py
from dataclasses import dataclass, field
from typing import List, Optional
from backend.core.enums import LotStatus


@dataclass
class Lot:
    """单批次买入明细"""
    lot_index: int        # 批次序号：1/2/3
    open_price: float     # 买入价格（元/g）
    amount_g: float       # 买入克数
    open_ts: int          # 买入时间（毫秒时间戳）
    status: LotStatus = LotStatus.OPEN
    close_ts: Optional[int] = None
    close_price: Optional[float] = None
    close_reason: Optional[str] = None


class PortfolioPosition:
    """
    T仓组合持仓管理。
    一轮交易 = 从第一笔买入到全部平仓。
    所有止盈/止损以组合整体盈亏率为基准，不对单笔单独止损。
    """

    def __init__(self, round_id: int):
        self.round_id = round_id        # 关联 positions.id
        self.lots: List[Lot] = []
        self.tp1_done: bool = False     # 第1次止盈是否已执行
        self.tp2_done: bool = False     # 第2次止盈是否已执行
        self._total_amount_g: float = 0.0
        self._total_cost: float = 0.0   # 总成本 = Σ(买入价 × 买入量)

    # ── 属性 ─────────────────────────────────────────────

    @property
    def total_amount_g(self) -> float:
        return self._total_amount_g

    @property
    def total_cost(self) -> float:
        return self._total_cost

    @property
    def avg_cost(self) -> float:
        """加权平均成本价（元/g）"""
        if self._total_amount_g == 0:
            return 0.0
        return self._total_cost / self._total_amount_g

    def is_empty(self) -> bool:
        return self._total_amount_g == 0.0

    # ── 核心计算 ─────────────────────────────────────────

    def pnl_pct(self, current_price: float) -> float:
        """
        T仓整体浮盈浮亏率。
        pnl_pct = (当前市值 - 总成本) / 总成本
        """
        if self._total_cost == 0:
            return 0.0
        market_value = current_price * self._total_amount_g
        return (market_value - self._total_cost) / self._total_cost

    # ── 建仓 / 加仓 ──────────────────────────────────────

    def add_lot(self, lot_index: int, price: float, amount_g: float, ts: int) -> Lot:
        """买入一批，返回 Lot 对象（调用方负责写库）"""
        lot = Lot(
            lot_index=lot_index,
            open_price=price,
            amount_g=amount_g,
            open_ts=ts,
        )
        self.lots.append(lot)
        self._total_amount_g += amount_g
        self._total_cost += price * amount_g
        return lot

    @property
    def lot_count(self) -> int:
        """当前批次数（含未平仓）"""
        return sum(1 for l in self.lots if l.status == LotStatus.OPEN)

    # ── 减仓 / 清仓 ──────────────────────────────────────

    def reduce(self, ratio: float, close_price: float, ts: int) -> float:
        """
        按比例减仓，返回实际卖出克数。
        按平均成本比例减少总成本，不对单笔单独计算。
        ratio: 0.0 ~ 1.0，相对当前持仓量的卖出比例
        """
        sold_g = self._total_amount_g * ratio
        cost_removed = self._total_cost * ratio
        self._total_amount_g -= sold_g
        self._total_cost -= cost_removed
        # 四舍五入避免浮点误差积累
        self._total_amount_g = round(self._total_amount_g, 4)
        self._total_cost = round(self._total_cost, 4)
        return sold_g

    def clear(self, close_price: float, ts: int) -> float:
        """全部清仓，返回实际卖出克数"""
        sold_g = self._total_amount_g
        self._total_amount_g = 0.0
        self._total_cost = 0.0
        for lot in self.lots:
            if lot.status == LotStatus.OPEN:
                lot.status = LotStatus.CLOSED
                lot.close_ts = ts
                lot.close_price = close_price
        return sold_g

    # ── 止盈状态 ─────────────────────────────────────────

    def mark_tp1(self):
        self.tp1_done = True

    def mark_tp2(self):
        self.tp2_done = True
```

```python
# backend/signals/buy_signal.py
from dataclasses import dataclass
from typing import Optional
from backend.core.enums import MarketState, SignalType
from backend.risk.position import PortfolioPosition
from backend import config


@dataclass
class BuySignal:
    signal_type: SignalType   # BUY（初始建仓）或 ADD_LOT（加仓）
    amount_g: float
    reason: str


def check_buy_signal(
    ctx,
    portfolio: PortfolioPosition,
    circuit_breaker_active: bool,
    last_buy_price: Optional[float] = None,
) -> Optional[BuySignal]:
    """
    检查是否触发建仓/加仓信号。
    返回 BuySignal 或 None（无信号）。
    last_buy_price: 上一批次的买入价，用于判断加仓间距
    """
    if circuit_breaker_active:
        return None

    price = ctx.current_price
    pnl = portfolio.pnl_pct(price)

    # 停止加仓：浮亏超过阈值
    if pnl <= config.STOP_ADD_LOSS_PCT:
        return None

    market_state = ctx.market_state

    # ── 震荡模式：初始建仓 ────────────────────────────────
    if market_state in (MarketState.OSCILLATION, MarketState.DECAY):
        return _check_oscillation_buy(ctx, portfolio, price, pnl, last_buy_price)

    # ── 趋势上涨模式 ──────────────────────────────────────
    if market_state == MarketState.TREND_UP:
        return _check_trend_buy(ctx, portfolio, price, pnl, last_buy_price)

    # TREND_DOWN / 其他：禁止建仓
    return None


def _check_oscillation_buy(ctx, portfolio, price, pnl, last_buy_price):
    bb_lower = ctx.indicators_5m.bb_lower
    atr = ctx.indicators_5m.atr

    if portfolio.is_empty():
        # 第1批：价格触及布林下轨
        if price <= bb_lower:
            return BuySignal(
                signal_type=SignalType.BUY,
                amount_g=config.LOT1_AMOUNT_G,
                reason=f"价格{price:.2f}触及布林下轨{bb_lower:.2f}，初始建仓",
            )
        return None

    if portfolio.total_amount_g >= config.T_MAX_AMOUNT_G:
        return None  # 已满仓

    # 第2/3批：价格从上次买入再跌 1×ATR
    if last_buy_price is None:
        return None

    required_drop = config.ATR_ADD_LOT_MULTIPLIER * atr
    if price <= last_buy_price - required_drop:
        lot_count = portfolio.lot_count
        if lot_count == 1:
            return BuySignal(
                signal_type=SignalType.ADD_LOT,
                amount_g=config.LOT2_AMOUNT_G,
                reason=f"价格从{last_buy_price:.2f}跌{required_drop:.2f}，第2批加仓",
            )
        if lot_count == 2:
            return BuySignal(
                signal_type=SignalType.ADD_LOT,
                amount_g=config.LOT3_AMOUNT_G,
                reason=f"价格从{last_buy_price:.2f}跌{required_drop:.2f}，第3批加仓",
            )
    return None


def _check_trend_buy(ctx, portfolio, price, pnl, last_buy_price):
    ema20 = ctx.indicators_4h.ema20
    ema60 = ctx.indicators_4h.ema60
    rsi = ctx.indicators_5m.rsi
    atr = ctx.indicators_5m.atr

    # 中期趋势确认
    if ema20 <= ema60:
        return None

    if portfolio.is_empty():
        # 初始建仓：RSI回调至≤40
        if rsi <= 40:
            return BuySignal(
                signal_type=SignalType.BUY,
                amount_g=config.LOT1_AMOUNT_G,
                reason=f"趋势上涨，RSI={rsi:.1f}回调超卖，初始建仓",
            )
        return None

    # 加仓规则同震荡模式
    return _check_oscillation_buy(ctx, portfolio, price, pnl, last_buy_price)
```

```python
from dataclasses import dataclass
from typing import Optional
from backend.core.enums import ExitReason, MarketState
from backend.risk.position import PortfolioPosition
from backend import config


@dataclass
class ExitSignal:
    exit_reason: ExitReason
    sell_ratio: float   # 0.0~1.0，相对当前持仓的卖出比例
    reason: str


def check_exit_signal(
    portfolio: PortfolioPosition,
    current_price: float,
    ctx,
) -> Optional[ExitSignal]:
    """
    检查是否触发止损/清仓信号。
    优先级：趋势清仓 > 亏损清仓 > 亏损减半
    """
    if portfolio.is_empty():
        return None

    # ── 趋势转空：立即清仓 ────────────────────────────────
    if ctx.market_state == MarketState.TREND_DOWN:
        return ExitSignal(
            exit_reason=ExitReason.TREND_CLEAR,
            sell_ratio=1.0,
            reason="趋势转空（-DI>+DI且ADX上升），立即清仓",
        )

    pnl = portfolio.pnl_pct(current_price)

    # ── 亏损-3.5%：全部清仓 ───────────────────────────────
    if pnl <= config.CLEAR_ALL_LOSS_PCT:
        return ExitSignal(
            exit_reason=ExitReason.STOP_LOSS_CLEAR,
            sell_ratio=1.0,
            reason=f"T仓浮亏{pnl:.2%}≥{config.CLEAR_ALL_LOSS_PCT:.2%}，全部清仓",
        )

    # ── 亏损-2.5%：强制减半 ───────────────────────────────
    if pnl <= config.FORCE_HALF_LOSS_PCT:
        return ExitSignal(
            exit_reason=ExitReason.STOP_LOSS_HALF,
            sell_ratio=0.50,
            reason=f"T仓浮亏{pnl:.2%}≥{config.FORCE_HALF_LOSS_PCT:.2%}，强制减仓50%",
        )

    # 亏损-1.5%以内：无减仓动作（停止加仓由 buy_signal 处理）
    return None
```

```python
from dataclasses import dataclass
from typing import Optional
from backend.core.enums import ExitReason
from backend.risk.position import PortfolioPosition
from backend import config


@dataclass
class SellSignal:
    exit_reason: ExitReason
    sell_ratio: float   # 相对当前持仓的卖出比例
    reason: str


def check_sell_signal(
    portfolio: PortfolioPosition,
    ctx,
) -> Optional[SellSignal]:
    """
    检查是否触发止盈信号。
    优先级：tp1 > tp2 > 追踪止盈（EMA跌破）
    """
    if portfolio.is_empty():
        return None

    price = ctx.current_price
    ema20 = ctx.indicators_5m.ema20
    pnl = portfolio.pnl_pct(price)

    # ── 第1次止盈：盈利≥0.6%，卖出60% ──────────────────────
    if not portfolio.tp1_done and pnl >= config.TAKE_PROFIT_1_PCT:
        return SellSignal(
            exit_reason=ExitReason.TAKE_PROFIT_1,
            sell_ratio=config.TAKE_PROFIT_1_SELL_RATIO,
            reason=f"T仓整体盈利{pnl:.2%}≥{config.TAKE_PROFIT_1_PCT:.2%}，卖出60%",
        )

    # ── 第2次止盈：盈利≥1.2%，卖出20% ──────────────────────
    if portfolio.tp1_done and not portfolio.tp2_done and pnl >= config.TAKE_PROFIT_2_PCT:
        return SellSignal(
            exit_reason=ExitReason.TAKE_PROFIT_2,
            sell_ratio=config.TAKE_PROFIT_2_SELL_RATIO,

            sell_ratio=1.0,
            reason=f"金价{price:.2f}跌破5分钟EMA20={ema20:.2f}，清空剩余持仓",
```

```python
def on_tick(self, price: float, ts: int):
    # 1. 熔断检查：熔断中只检查持仓止损，不开新仓
    cb_active = self.circuit_breaker.is_active()

    # 2. 更新 context（K线合成、指标计算）
    self.context.update(price, ts)

    if self.context.indicators_5m is None:
        return  # 数据不足，跳过

    # 3. 计算T仓整体盈亏率（记录到 context 供 WebSocket 推送）
    pnl_pct = self.portfolio.pnl_pct(price)

    # 4. 止损检查（优先级最高）
    exit_signal = check_exit_signal(self.portfolio, price, self.context)
    if exit_signal:
        self._execute_exit(exit_signal, price, ts)

    # 5. 止盈检查
    if not self.portfolio.is_empty():
        sell_signal = check_sell_signal(self.portfolio, self.context)
        if sell_signal:
            self._execute_sell(sell_signal, price, ts)

    # 6. 建仓/加仓检查
    if not cb_active:
        buy_signal = check_buy_signal(
            self.context, self.portfolio,
            circuit_breaker_active=cb_active,
            last_buy_price=self._last_buy_price,
        )
        if buy_signal:
            self._execute_buy(buy_signal, price, ts)

    # 7. 检查是否触发新的熔断条件
    self.circuit_breaker.check_and_trigger(price, ts, self.context)

    # 8. 推送 WebSocket 更新
    self._broadcast(price, ts, pnl_pct)
```

```python
def _execute_buy(self, signal, price: float, ts: int):
    """执行建仓/加仓"""
    from backend.core.enums import SignalType
    if self.portfolio.is_empty():
        # 新的一轮，创建 position 记录
        self._round_id_counter += 1
        self.portfolio = PortfolioPosition(round_id=self._round_id_counter)
        self._save_position_open(ts)

    lot = self.portfolio.add_lot(
        lot_index=self.portfolio.lot_count,
        price=price,
        amount_g=signal.amount_g,
        ts=ts,
    )
    self._last_buy_price = price
    self._save_lot(lot)
    self._save_signal(signal.signal_type, price, signal.amount_g, signal.reason, ts)

def _execute_sell(self, signal, price: float, ts: int):
    """执行止盈减仓/清仓"""
    from backend.core.enums import ExitReason
    sold_g = self.portfolio.reduce(signal.sell_ratio, price, ts) \
        if signal.sell_ratio < 1.0 else self.portfolio.clear(price, ts)

    if signal.exit_reason == ExitReason.TAKE_PROFIT_1:
        self.portfolio.mark_tp1()
    elif signal.exit_reason == ExitReason.TAKE_PROFIT_2:
        self.portfolio.mark_tp2()

    self._save_signal(signal.exit_reason, price, sold_g, signal.reason, ts)
    if self.portfolio.is_empty():
        self._save_position_close(price, ts, signal.exit_reason)
        self._last_buy_price = None

def _execute_exit(self, signal, price: float, ts: int):
    """执行止损减仓/清仓"""
    if signal.sell_ratio < 1.0:
        sold_g = self.portfolio.reduce(signal.sell_ratio, price, ts)
    else:
        sold_g = self.portfolio.clear(price, ts)
    self._save_signal(signal.exit_reason, price, sold_g, signal.reason, ts)
    if self.portfolio.is_empty():
        self._save_position_close(price, ts, signal.exit_reason)
        self._last_buy_price = None
```
